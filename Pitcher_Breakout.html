Building the model

In[112]:
from sklearn.model_selection import train_test_split
from xgboost import XGBRegressor

features = ['k_percent', 'bb_percent', 'wobadiff', 'hard_hit_percent', 'in_zone_percent', 'whiff_percent', 'groundballs_percent', 'flyballs_percent']  
X = Pitching_2023_Filtered[features]
y = Pitching_2023_Filtered['ERA_2024']

X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.2, random_state=21)

xgb_model = XGBRegressor(
    n_estimators=500,
    max_depth=4,
    learning_rate=0.05,
    subsample=0.8,
    colsample_bytree=0.8,
    random_state=21
)

xgb_model.fit(X_train, y_train)
